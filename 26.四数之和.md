/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[][]}
 */
var fourSum = function(nums, target) {
    let result = [];
    nums.sort((a,b) => a -b)

    for(let i = 0; i < nums.length; i++) {
        let a = nums[i];
        if(target > 0 && a > target) break;
        if(a === nums[i - 1]) continue;
        for(let j = i + 1; j < nums.length; j++) {
            let b = nums[j];
            let left = j + 1;
            let right = nums.length - 1;
            if(target > 0 && a + b > target) break;
            if(j > i + 1 && b === nums[j - 1]) continue;
            while(left < right) {
                let c = nums[left];
                let d = nums[right];
                let sum = a + b + c + d;
                if(sum > target) {
                    right--;
                } else if (sum < target) {
                    left++;
                } else {
                    result.push([a,b,c,d]);
                    while(nums[left] === nums[left + 1]) {
                        left++;
                    }
                    while(nums[right] === nums[right - 1]) {
                        right--;
                    }
                    left++;
                    right--;
                }
            }
        }
    }
    return result;
};
