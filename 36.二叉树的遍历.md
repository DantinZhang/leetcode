#### 2.中序遍历
```javascrtip
var inorderTraversal = function(root) {
    let res = [];

    function search(root) {
        if(root === null) return;
        search(root.left);
        res.push(root.val);
        search(root.right);
    }

    search(root);
    return res;
};
```
