//双指针法，空间换时间，时间复杂度O(N)
 var sortedSquares = function(nums) {
    let i = 0;   //一个指针指向头部
    let j = nums.length - 1;  //一个指针指向尾部
    let result = [];
   //注意循环停止的条件，当i = j时，指针相撞，指向的应该是最小的元素，循环应继续
    while(i<=j) {
        if(Math.pow(nums[i],2) < Math.pow(nums[j],2)) {
            result.unshift(Math.pow(nums[j],2));
            j--;
        } else {
            result.unshift(Math.pow(nums[i],2));
            i++;
        }
    }
    return result
};
