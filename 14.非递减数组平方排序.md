 var sortedSquares = function(nums) {
    let i = 0;
    let j = nums.length - 1;
    let result = []
    while(i<=j) {
        if(Math.pow(nums[i],2) < Math.pow(nums[j],2)) {
            result.unshift(Math.pow(nums[j],2));
            j--;
        } else {
            result.unshift(Math.pow(nums[i],2));
            i++;
        }
    }
    return result
};
